# ToonTown MultiTool v2.0 — Python dependencies
# Install with: pip install -r requirements.txt
#
# Dependency policy:
#   pip installs the newest release compatible with the running Python via
#   each wheel's Requires-Python metadata, so modern systems get modern deps
#   automatically while older ones (Python 3.9 / Debian 11) still resolve.
#   Keep the >= floors at the lowest version that ships Python 3.9 wheels AND
#   has the APIs this app uses. Add an upper cap ONLY when a known newer
#   release breaks Python 3.9 — always behind a python_version marker, never
#   a blanket pin. The portability CI matrix (.github/workflows/portability.yml)
#   is the source of truth: it runs pip install across Python 3.9–3.12.

# Core — required on all platforms
# Pin (legacy Python): PySide6 6.9.x bundles a Qt6.9 build with a font-engine
# null-pointer regression in QFontEngineFT::shouldUseDesignMetrics that segfaults
# during first-launch font rendering on Linux Mint 22 (Ubuntu 24.04 base). Locking
# below 6.9 for Pythons that still receive 6.8 wheels. The 6.8 wheels' Requires-Python
# metadata caps at <3.13, so this pin only applies to Python 3.9-3.12.
PySide6>=6.8,<6.9; python_version < "3.13"

# Pin (modern Python): PySide6 6.10+ is required for Python 3.13/3.14 (6.8 wheels
# aren't built for those Pythons per their Requires-Python metadata). Skipping
# 6.9.x for the same NULL-pointer bug as above. Broad <7 range matches the
# existing major-version policy.
PySide6>=6.10,<7; python_version >= "3.13"
pynput>=1.7.6
requests>=2.28.0
keyring>=24.0.0
certifi>=2023.7.22
cryptography>=41.0.0

# Linux only — keyring backend dependencies (Secret Service for GNOME/Cinnamon, DBus for KWallet)
secretstorage>=3.3.0; sys_platform == "linux"
jeepney>=0.7.0; sys_platform == "linux"

# Linux only — Xlib input backend and XRes PID resolution
python-xlib>=0.33; sys_platform == "linux"

# macOS only — PyObjC bridges for CGEvent injection, CGWindowList discovery,
# CGEventTap capture (Quartz) and NSWorkspace/NSRunningApplication (Cocoa).
# Declared explicitly rather than relying on pynput's transitive PyObjC deps.
pyobjc-framework-Quartz>=10.0; sys_platform == "darwin"
pyobjc-framework-Cocoa>=10.0; sys_platform == "darwin"

# Linux only — YAML parsing for Bottles and Lutris config files
PyYAML>=6.0; sys_platform == "linux"

# Windows only — Win32 input backend and window management
pywin32>=306; sys_platform == "win32"

# Cross-platform process/file/network introspection. Used by:
#   - Windows: port->PID enumeration without spawning netstat (avoids
#     Smart App Control / Defender heuristics on the unsigned EXE).
#   - Linux: CC log discovery (utils/cc_log_discovery.py uses
#     process_iter + Process.open_files to find an externally-launched
#     CC log) and the settings-tab Detect button.
psutil>=5.9

# Build-time dependency for PyInstaller-based packaging (Linux AppImage + Windows EXE).
pyinstaller>=6.0,<7
